\section{Intutive Definition of a Limit}

\begin{equation} 
    \lim_{x \to a} f(x) = L
\end{equation}

\begin{definition}
    {Definition of a Limit}
    The limit of a function $f(x)$ as $x$ approaches $a$ is $L$ if for every $\epsilon > 0$ there exists a $\delta > 0$ such that if $0 < |x - a| < \delta$, then $|f(x) - L| < \epsilon$.
\end{definition}
